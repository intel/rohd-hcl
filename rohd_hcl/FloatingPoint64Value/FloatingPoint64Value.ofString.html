factory FloatingPoint64Value.ofString(String fp) {
  final s = fp.split(' ');
  assert(s.length == 3, 'Wrong FloatingPointValue string length ${s.length}');
  return FloatingPoint64Value.ofStrings(s[0], s[1], s[2]);
}
